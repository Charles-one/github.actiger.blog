3. Ingress Nginx的配置my-nginx.yml


apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  namespace: nginx-space
  name: my-nginx
spec:
  tls:
  - hosts:
    - weini.actiger.com
    - www.actiger.com
    secretName: ingress-secret
  - hosts:
    - actiger.com
    secretName: ingress-top-secret
  rules:
  - host: weini.actiger.com
    http:
      paths:
      - backend:
          serviceName: my-nginx
          servicePort: 80
  - host: www.actiger.com
    http:
      paths:
      - backend:
          serviceName: my-nginx
          servicePort: 80
  - host: actiger.com
    http:
      paths:
      - backend:
          serviceName: my-nginx
          servicePort: 80
  - host: test.actiger.com
    http:
      paths:
      - backend:
          serviceName: my-nginx
          servicePort: 80
